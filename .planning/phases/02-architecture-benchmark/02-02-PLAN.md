import subprocess
import os

def export_to_onnx(model, tokenizer, model_name, output_dir="research/models/benchmark_tmp"):
    """Export PyTorch model to ONNX format.
    CRITICAL: dynamo=False required (Pitfall 2 from RESEARCH.md).
    CRITICAL: model.cpu() required (Pitfall 8 from RESEARCH.md).
    """
    model.eval()
    model_cpu = model.cpu()

    dummy = tokenizer(
        "benchmark test text for onnx export",
        return_tensors="pt",
        padding="max_length",
        max_length=128,
        truncation=True,
    )

    onnx_path = os.path.join(output_dir, f"{model_name.lower().replace(' ', '_').replace('-', '_')}.onnx")
    os.makedirs(os.path.dirname(onnx_path), exist_ok=True)

    torch.onnx.export(
        model_cpu,
        tuple(dummy.values()),
        onnx_path,
        input_names=list(dummy.keys()),
        output_names=["logits"],
        dynamic_axes={n: {0: "batch", 1: "seq"} for n in dummy.keys()}
        | {"logits": {0: "batch"}},
        opset_version=14,
        dynamo=False,  # CRITICAL: legacy exporter required (Pitfall 2)
    )

    file_size_mb = os.path.getsize(onnx_path) / (1024 * 1024)
    print(f"{model_name} ONNX exported: {onnx_path} ({file_size_mb:.1f} MB)")

    # Sanity check: ONNX file must be > 1MB for models with >10M params
    assert file_size_mb > 1.0, f"ONNX file too small ({file_size_mb:.1f} MB) -- dynamo exporter may have been used"

    return onnx_path

# Export all 3 models
# NOTE: trainer objects (trainer_mb, trainer_tb, trainer_el) and tokenizers
# (tokenizer_mb, tokenizer_tb, tokenizer_el) are from Plan 01 training cells

onnx_paths = {}
for name, trainer_obj, tokenizer_obj in [
    ("MobileBERT", trainer_mb, tokenizer_mb),
    ("TinyBERT-4", trainer_tb, tokenizer_tb),
    ("ELECTRA-small", trainer_el, tokenizer_el),
]:
    onnx_paths[name] = export_to_onnx(trainer_obj.model, tokenizer_obj, name)
    print(f"  Input names: {list(tokenizer_obj('test', return_tensors='pt').keys())}")

def convert_onnx_to_tflite(onnx_path, model_name, output_dir="research/models/benchmark_tmp"):
    """Convert ONNX model to TFLite via onnx2tf.
    D-02 deviation: Using ONNX->onnx2tf instead of optimum==1.27.0
    (optimum 2.1.0 installed, TFLite export removed in optimum 2.0).
    """
    tflite_dir = os.path.join(output_dir, f"{model_name.lower().replace(' ', '_').replace('-', '_')}_tflite")
    os.makedirs(tflite_dir, exist_ok=True)

    cmd = [
        ".venv/bin/python3", "-m", "onnx2tf",
        "-i", onnx_path,
        "-o", tflite_dir,
        "-nuo",  # non-unique name output to avoid name conflicts
    ]

    print(f"\nConverting {model_name} to TFLite...")
    result = subprocess.run(cmd, capture_output=True, text=True, timeout=300)

    if result.returncode != 0:
        print(f"  STDERR: {result.stderr[-500:]}")
        raise RuntimeError(f"onnx2tf failed for {model_name}: {result.stderr[-200:]}")

    # Find the .tflite file (onnx2tf outputs multiple files; pick the float32 one)
    tflite_files = sorted(
        [f for f in os.listdir(tflite_dir) if f.endswith(".tflite")],
        key=lambda f: os.path.getsize(os.path.join(tflite_dir, f)),
        reverse=True,  # Largest file is the full model
    )
    assert len(tflite_files) > 0, f"No .tflite files produced for {model_name}"

    tflite_path = os.path.join(tflite_dir, tflite_files[0])
    size_mb = os.path.getsize(tflite_path) / (1024 * 1024)
    print(f"  {model_name} TFLite: {tflite_path} ({size_mb:.1f} MB)")

    return tflite_path, size_mb

tflite_paths = {}
tflite_sizes = {}
for name in ["MobileBERT", "TinyBERT-4", "ELECTRA-small"]:
    try:
        path, size = convert_onnx_to_tflite(onnx_paths[name], name)
        tflite_paths[name] = path
        tflite_sizes[name] = size
    except Exception as e:
        print(f"  {name} TFLite conversion FAILED: {e}")
        tflite_paths[name] = None
        tflite_sizes[name] = -1
```

from ai_edge_litert import interpreter as litert
# NOTE: Do NOT use tf.lite.Interpreter — it is deprecated and crashes
# on numpy 2.4 (Pitfall 3 from RESEARCH.md)

def validate_tflite(tflite_path, model_name):
    """Validate TFLite model: load, check inputs/outputs, run inference.
    Returns (passed, input_details, output_details) or (False, None, None) if disqualified.
    """
    try:
        # Load with ONLY built-in ops (no flex delegate)
        interp = litert.Interpreter(model_path=tflite_path)
        interp.allocate_tensors()

        input_details = interp.get_input_details()
        output_details = interp.get_output_details()

        print(f"\n{model_name} TFLite Validation:")
        print(f"  Inputs:")
        for inp in input_details:
            print(f"    {inp['name']}: shape={inp['shape']}, dtype={inp['dtype']}")
        print(f"  Outputs:")
        for out in output_details:
            print(f"    {out['name']}: shape={out['shape']}, dtype={out['dtype']}")

        # Run test inference with dummy data
        for inp in input_details:
            interp.set_tensor(
                inp['index'],
                np.ones(inp['shape'], dtype=inp['dtype'])
            )
        interp.invoke()

        # Check output shape is [1, 2] (binary classification)
        output = interp.get_tensor(output_details[0]['index'])
        print(f"  Output shape: {output.shape}, values: {output}")
        assert output.shape[-1] == 2, f"Expected 2-class output, got shape {output.shape}"

        print(f"  {model_name}: PASSED -- standard TFLite ops only")
        return True, input_details, output_details

    except Exception as e:
        print(f"  {model_name}: DISQUALIFIED -- requires SELECT_TF_OPS or other issue: {e}")
        return False, None, None

tflite_validation = {}
for name in ["MobileBERT", "TinyBERT-4", "ELECTRA-small"]:
    if tflite_paths.get(name):
        passed, inp_det, out_det = validate_tflite(tflite_paths[name], name)
        tflite_validation[name] = {
            "passed": passed,
            "inputs": [{"name": d["name"], "shape": d["shape"].tolist(), "dtype": str(d["dtype"])} for d in inp_det] if inp_det else [],
            "outputs": [{"name": d["name"], "shape": d["shape"].tolist(), "dtype": str(d["dtype"])} for d in out_det] if out_det else [],
        }
    else:
        tflite_validation[name] = {"passed": False, "inputs": [], "outputs": []}
        print(f"{name}: SKIPPED (TFLite conversion failed)")

def measure_latency(tflite_path, model_name, n_warmup=10, n_runs=100):
    """Measure TFLite inference latency on desktop.
    Per D-08: Desktop interpreter only, sufficient for ranking.
    Per D-09: On-device measurement deferred to Phase 6.
    """
    interp = litert.Interpreter(model_path=tflite_path)
    interp.allocate_tensors()

    input_details = interp.get_input_details()

    # Set up input data (random int32 token IDs)
    for inp in input_details:
        interp.set_tensor(
            inp['index'],
            np.random.randint(0, 1000, size=inp['shape']).astype(inp['dtype'])
        )

    # Warm up
    for _ in range(n_warmup):
        interp.invoke()

    # Measure
    latencies = []
    for _ in range(n_runs):
        start = time.perf_counter()
        interp.invoke()
        elapsed = (time.perf_counter() - start) * 1000  # ms
        latencies.append(elapsed)

    p50 = np.median(latencies)
    p95 = np.percentile(latencies, 95)
    print(f"{model_name} Latency: p50={p50:.1f}ms, p95={p95:.1f}ms (100 runs)")
    return {"p50_ms": round(p50, 1), "p95_ms": round(p95, 1)}

tflite_latencies = {}
for name in ["MobileBERT", "TinyBERT-4", "ELECTRA-small"]:
    if tflite_paths.get(name) and tflite_validation.get(name, {}).get("passed"):
        tflite_latencies[name] = measure_latency(tflite_paths[name], name)
    else:
        tflite_latencies[name] = {"p50_ms": -1, "p95_ms": -1}
        print(f"{name}: SKIPPED latency (TFLite validation failed)")
```
