    """Run all preflight checks before main generation. Returns timing info."""
    results = {}

    # Check 1: Ollama availability + model verification (review item 2)

        if OLLAMA_MODEL not in models and f"{OLLAMA_MODEL}:latest" not in [m.split(":")[0] + ":latest" if ":" not in m else m for m in models]:
            # Try matching without tag
            found = any(OLLAMA_MODEL.split(":")[0] in m for m in models)
            if not found:
                sys.exit(f"[PREFLIGHT FAIL] {OLLAMA_MODEL} not found in Ollama. Available: {models}\nRun: ollama pull {OLLAMA_MODEL}")
        print(f"[PREFLIGHT OK] Ollama running, {OLLAMA_MODEL} available")
    except requests.ConnectionError:
        sys.exit("[PREFLIGHT FAIL] Ollama not running at localhost:11434. Start with: ollama serve")

    # Check 2: Ollama hardware benchmark (review item 3)
    print("[PREFLIGHT] Benchmarking Ollama inference speed (2 test samples)...")
    import time
    ollama_times = []
    for i in range(2):
        start = time.time()
        test_sample = generate_ollama("Write a short test message about the weather.", OLLAMA_MODEL)
        elapsed = time.time() - start
        ollama_times.append(elapsed)

    avg_time = sum(ollama_times) / len(ollama_times)
    ollama_total_estimate = avg_time * (TOTAL_SCAM_TARGET * OLLAMA_SHARE + SAFE_TARGET * OLLAMA_SHARE)
    print(f"[PREFLIGHT] Ollama avg: {avg_time:.1f}s/sample. Estimated total for {int(TOTAL_SCAM_TARGET * OLLAMA_SHARE)} samples: {ollama_total_estimate/3600:.1f} hours")
    if ollama_total_estimate > 86400:  # > 24 hours
        print(f"[WARNING] Ollama generation will take >{ollama_total_estimate/3600:.0f} hours.")
        print("  Consider reducing OLLAMA_SHARE to 0.15 (document deviation from D-05)")
    results["ollama_avg_seconds"] = avg_time

    # Check 3: Gemini preflight test (review item 4)
    print("[PREFLIGHT] Testing Gemini with 5 vectors (5 samples)...")
    test_vectors = ["crypto_investment", "romance_grooming", "tech_support", "phishing", "lottery_reward"]
    gemini_failures = 0
    for vector in test_vectors:

        sys.exit(f"[PREFLIGHT FAIL] Gemini rejected {gemini_failures}/5 test prompts. Revise indirect prompting strategy before proceeding.")
    print(f"[PREFLIGHT OK] Gemini: {5 - gemini_failures}/5 test prompts succeeded")

    # Check 4: GEMINI_API_KEY
    if not os.environ.get("GEMINI_API_KEY"):
        sys.exit("[PREFLIGHT FAIL] GEMINI_API_KEY not set. Get key from https://aistudio.google.com/apikey")

    # Check 5: Holdout exists
    if not HOLDOUT_PATH.exists():
        sys.exit(f"[PREFLIGHT FAIL] Holdout not found at {HOLDOUT_PATH}. Run Plan 01 first.")

    return results

4. **Gemini generation function with exponential backoff (addresses review item 10):**
```python
from google import genai
from google.genai import types
from pydantic import BaseModel
import time

class ScamSample(BaseModel):
    text: str
    label: str
    vector: str
    channel: str

def generate_gemini(prompt: str, client: genai.Client, max_retries: int = 3) -> dict | None:
    """Generate one sample via Gemini 2.5 Flash with exponential backoff."""
    for attempt in range(max_retries):
        try:
            response = client.models.generate_content(
                model="gemini-2.5-flash",
                contents=prompt,
                config=types.GenerateContentConfig(
                    response_mime_type="application/json",
                    response_json_schema=ScamSample.model_json_schema(),
                ),
            )
            sample = json.loads(response.text)
            sample["source"] = "gemini-2.5-flash"
            return sample
        except Exception as e:
            if "429" in str(e) or "RESOURCE_EXHAUSTED" in str(e):
                wait = 2 ** attempt  # 1s, 2s, 4s
                print(f"  Rate limited, retrying in {wait}s...")
                time.sleep(wait)
            else:
                print(f"Gemini error: {e}")
                return None
    print(f"Gemini failed after {max_retries} retries")
    return None
```

5. **Ollama generation function with JSON parse error handling (addresses review concern):**
```python
import requests

def generate_ollama(prompt: str, model: str = "llama3.1:8b") -> dict | None:
    """Generate one sample via local Ollama. Returns None on failure."""
    try:
        response = requests.post(
            "http://localhost:11434/api/generate",
            json={"model": model, "prompt": prompt, "stream": False, "format": "json"},
            timeout=180,  # CPU inference is slow — allow 3 minutes per sample
        )
        response.raise_for_status()
        raw = response.json()["response"]
        try:
            sample = json.loads(raw)
        except json.JSONDecodeError:
            print(f"  Ollama returned non-JSON: {raw[:100]}...")
            return None
        sample["source"] = "llama3.1:8b"
        return sample
    except requests.Timeout:
        print(f"  Ollama timeout (180s) — sample skipped")
        return None
    except Exception as e:
        print(f"Ollama error: {e}")
        return None
```
